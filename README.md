# Inicializar un proyecto .git nuevo o vacio 

    $ git init

# Saber el estado de nuestro proyecto git

    $ git status

# Preparar un archivo para seguimiento (staging)

    $ git add <file>
    $ git add .
    $ git add -A
    $ git add *.md

# Guadamos nuestros archivos preparados en el repositorio (commit)

    $ git commit -m "Pagina Principal Home"
